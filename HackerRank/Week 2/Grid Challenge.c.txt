#include <stdio.h>
#include <string.h>

void sortRow(char s[], int n) {
    for (int i = 0; i < n; i++) {
        for (int j = i + 1; j < n; j++) {
            if (s[i] > s[j]) {
                char t = s[i];
                s[i] = s[j];
                s[j] = t;
            }
        }
    }
}

int main() {
    int t;
    scanf("%d", &t);
    while (t--) {
        int n;
        scanf("%d", &n);
        char grid[n][n + 1];
        for (int i = 0; i < n; i++) {
            scanf("%s", grid[i]);
            sortRow(grid[i], n);
        }
        int ok = 1;
        for (int col = 0; col < n && ok; col++) {
            for (int row = 0; row < n - 1; row++) {
                if (grid[row][col] > grid[row + 1][col]) {
                    ok = 0;
                    break;
                }
            }
        }
        if (ok) printf("YES\n");
        else printf("NO\n");
    }
    return 0;
}
