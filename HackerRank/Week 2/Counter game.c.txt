#include <stdio.h>

int isPowerOfTwo(unsigned long long n) {
    return (n && !(n & (n - 1)));
}

unsigned long long highestPowerOfTwo(unsigned long long n) {
    unsigned long long p = 1ULL;
    while (p << 1 <= n) p <<= 1;
    return p;
}

const char* counterGame(unsigned long long n) {
    int turns = 0; // 0 = Louise, 1 = Richard
    while (n > 1) {
        if (isPowerOfTwo(n)) n /= 2;
        else n -= highestPowerOfTwo(n);
        turns ^= 1; // switch player
    }
    return turns ? "Louise" : "Richard";
}

int main() {
    int t;
    scanf("%d", &t);
    while (t--) {
        unsigned long long n;
        scanf("%llu", &n);
        printf("%s\n", counterGame(n));
    }
    return 0;
}
