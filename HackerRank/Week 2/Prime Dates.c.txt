#include <stdio.h>

int isLeap(int year) {
    if (year % 400 == 0) return 1;
    if (year % 100 == 0) return 0;
    if (year % 4 == 0) return 1;
    return 0;
}

void nextDate(int *d, int *m, int *y) {
    int monthDays[] = {0,31,28,31,30,31,30,31,31,30,31,30,31};
    int days = monthDays[*m];
    if (*m == 2 && isLeap(*y)) days = 29;

    (*d)++;
    if (*d > days) {
        *d = 1;
        (*m)++;
    }
    if (*m > 12) {
        *m = 1;
        (*y)++;
    }
}

int findLuckyDates(int d1, int m1, int y1, int d2, int m2, int y2) {
    int count = 0;
    while (1) {
        long long num = (long long)d1 * 1000000 + (long long)m1 * 10000 + y1;
        if (num % 4 == 0 || num % 7 == 0) count++;

        if (d1 == d2 && m1 == m2 && y1 == y2) break;

        nextDate(&d1, &m1, &y1);
    }
    return count;
}

int main() {
    int d1, m1, y1, d2, m2, y2;
    scanf("%d-%d-%d", &d1, &m1, &y1);
    scanf("%d-%d-%d", &d2, &m2, &y2);

    printf("%d\n", findLuckyDates(d1, m1, y1, d2, m2, y2));
    return 0;
}
