#include <stdio.h>
#include <stdlib.h>
#include <limits.h>

int cmp(const void *a, const void *b) {
    long long x = *(long long*)a;
    long long y = *(long long*)b;
    if (x < y) return -1;
    if (x > y) return 1;
    return 0;
}

long long minmax(int n, int k, long long a[]) {
    qsort(a, n, sizeof(long long), cmp);
    long long m = LLONG_MAX;
    for (int i = 0; i <= n - k; i++) {
        long long u = a[i + k - 1] - a[i];
        if (u < m) m = u;
    }
    return m;
}

int main() {
    int n, k;
    scanf("%d", &n);
    scanf("%d", &k);
    long long a[n];
    for (int i = 0; i < n; i++) {
        scanf("%lld", &a[i]);}
    long long r = minmax(n, k, a);
    printf("%lld\n", r);
    return 0;
}
