#include <stdio.h>


int sockMerchant(int n, int a[]) {
    int f[101]={0};
    int p=0;
    
    for (int i=0;i<n;i++){
        f[a[i]]++;
    }
    for(int i=0;i<101;i++){
        p+=f[i]/2;
    }
    return p;
}

int main()
{
    int n;
    scanf("%d",&n);
    int a[n];
    for(int i=0;i<n;i++){
        scanf("%d",&a[i]);
    }
    int r=sockMerchant(n, a);
    printf("%d",r);
    return 0;
}
