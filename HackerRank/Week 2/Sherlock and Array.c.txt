#include <stdio.h>

char* balancedSums(int arr[], int n) {
    long long total = 0, left_sum = 0;
    for (int i = 0; i < n; i++) total += arr[i];
    
    for (int i = 0; i < n; i++) {
        if (left_sum == total - left_sum - arr[i]) {
            return "YES";
        }
        left_sum += arr[i];
    }
    return "NO";
}

int main() {
    int t;
    scanf("%d", &t);
    while (t--) {
        int n;
        scanf("%d", &n);
        int arr[n];
        for (int i = 0; i < n; i++) scanf("%d", &arr[i]);
        printf("%s\n", balancedSums(arr, n));
    }
    return 0;
}
