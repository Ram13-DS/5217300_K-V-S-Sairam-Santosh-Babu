#include <stdio.h>
#include <string.h>

long long digitSum(char n[]) {
    long long sum = 0;
    for (int i = 0; n[i] != '\0'; i++) {
        sum += (n[i] - '0');
    }
    return sum;
}

int superDigitCalc(long long num) {
    if (num < 10) return (int)num;
    long long sum = 0;
    while (num > 0) {
        sum += num % 10;
        num /= 10;
    }
    return superDigitCalc(sum);
}

int superDigit(char n[], int k) {
    long long sum = digitSum(n);
    sum *= k;
    return superDigitCalc(sum);
}

int main() {
    char n[100001];
    int k;
    scanf("%s %d", n, &k);
    printf("%d\n", superDigit(n, k));
    return 0;
}
