#include <stdio.h>
#include <stdlib.h>

int cmp(const void *a, const void *b) {
    return (*(int*)a - *(int*)b);
}

int main() {
    int n, k;
    scanf("%d %d", &n, &k);
    int x[n];
    for (int i = 0; i < n; i++) scanf("%d", &x[i]);
    qsort(x, n, sizeof(int), cmp);
    int i = 0, transmitters = 0;
    while (i < n) {
        transmitters++;
        int loc = x[i] + k;
        while (i < n && x[i] <= loc) i++;
        loc = x[i-1] + k;
        while (i < n && x[i] <= loc) i++;
    }
    printf("%d\n", transmitters);
    return 0;
}
