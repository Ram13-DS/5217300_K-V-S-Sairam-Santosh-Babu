#include <stdio.h>

void countingSort(int a[],int n) {
 int c[100]={0};
 for(int i=0;i<n;i++){
    c[a[i]]++;
 }
 for(int i=0;i<100;i++){
    printf("%d ",c[i]);
 }
}

int main()
{
    int n;
    scanf("%d",&n);
    int a[n];
    for(int i=0;i<n;i++){
        scanf("%d",&a[i]);
    }
    countingSort(a,n);
    return 0;
}
