#include <stdio.h>

int birthday(int s[], int n, int d, int m) {
    int c=0;
    for(int i=0;i<=n-m;i++){
        int s1=0;
        for(int j=i;j<i+m;j++){
            s1+=s[j];
        }       
        if (s1==d)
            c++;
    }
    return c;
    }
int main()
{
    int n;
    scanf("%d",&n);
    int s[n];
    for(int i=0;i<n;i++){
        scanf("%d",&s[i]);}
    int d,m;
    scanf("%d%d",&d,&m);
    int r=birthday(s,n,d,m);
    printf("%d",r);
    return 0;
}
