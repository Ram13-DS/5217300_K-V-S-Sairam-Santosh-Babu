#include <stdio.h>

int cp(int n, int a[]) {
    int c = 0;
    for (int i = 0; i < n; i++) {
        if (a[i] > 0)
            c += 1;
    }
    return c;
}

int cz(int n, int a[]) {
    int c = 0;
    for (int i = 0; i < n; i++) {
        if (a[i] == 0)
            c += 1;
    }
    return c;
}

int cn(int n, int a[]) {
    int c = 0;
    for (int i = 0; i < n; i++) {
        if (a[i] < 0)
            c += 1;
    }
    return c;
}

void plusMinus(int n, int a[]) {
    int x = cp(n, a);
    int y = cn(n, a);
    int z = cz(n, a);

    printf("%.6f\n", (float)x / n);
    printf("%.6f\n", (float)y / n);
    printf("%.6f\n", (float)z / n);
}

int main() {
    int n;
    scanf("%d", &n);

    int arr[n];  
    for (int i = 0; i < n; i++) {
        scanf("%d", &arr[i]);
    }

    plusMinus(n, arr);

    return 0;
}
