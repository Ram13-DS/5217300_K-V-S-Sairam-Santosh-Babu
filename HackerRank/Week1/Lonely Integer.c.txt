
#include <stdio.h>



int lonelyinteger(int n, int a[]) {
 for (int i=0;i<n;i++){
    int c=0;
    for(int j=0;j<n;j++){
        if (a[i]==a[j]){
            c++;
        }
    }
    if (c==1){
        return a[i];
    }
 }
 return -1;
}

int main()
{
    int n;
    scanf("%d",&n);
    int a[n];
    for(int i=0;i<n;i++){
        scanf("%d",&a[i]);
    }
    int r=lonelyinteger(n, a);
    printf("%d",r);
    return 0;
}
