#include <stdio.h>

void Rides(int q[], int n) {
    int b = 0;
    for (int i = 0; i < n; i++) {
        if (q[i] - (i + 1) > 2) {
            printf("Too chaotic\n");
            return;
        }
        for (int j = (q[i] - 2 > 0 ? q[i] - 2 : 0); j < i; j++) {
            if (q[j] > q[i]) b++;
        }
    }
    printf("%d\n", b);
}

int main() {
    int t;
    scanf("%d", &t);
    while (t--) {
        int n;
        scanf("%d", &n);
        int q[n];
        for (int i = 0; i < n; i++) scanf("%d", &q[i]);
        Rides(q, n);
    }
    return 0;
}
