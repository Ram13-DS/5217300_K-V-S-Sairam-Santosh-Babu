#include <stdio.h>

int main() {
    int n;
    scanf("%d", &n);
    int s[n];
    for (int i = 0; i < n; i++) scanf("%d", &s[i]);

    int a[n], u = 0;
    a[u++] = s[0];
    for (int i = 1; i < n; i++) {
        if (s[i] != s[i-1]) a[u++] = s[i];
    }

    int m;
    scanf("%d", &m);
    int b[m];
    for (int i = 0; i < m; i++) scanf("%d", &b[i]);

    int j = u - 1;
    for (int i = 0; i < m; i++) {
        while (j >= 0 && b[i] >= a[j]) j--;
        printf("%d\n", j + 2);
    }

    return 0;
}
