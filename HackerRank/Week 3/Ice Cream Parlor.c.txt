#include <stdio.h>
#include <stdlib.h>

void icecreamParlor(int money, int* cost, int n) {
    for (int i = 0; i < n - 1; i++) {
        for (int j = i + 1; j < n; j++) {
            if (cost[i] + cost[j] == money) {
                printf("%d %d\n", i + 1, j + 1);
                return;
            }
        }
    }
}

int main() {
    int t;
    scanf("%d", &t);

    while (t--) {
        int money, n;
        scanf("%d", &money);
        scanf("%d", &n);

        int* cost = (int*)malloc(n * sizeof(int));
        for (int i = 0; i < n; i++) {
            scanf("%d", &cost[i]);
        }

        icecreamParlor(money, cost, n);
        free(cost);
    }

    return 0;
}
