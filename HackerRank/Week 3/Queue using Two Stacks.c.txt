#include <stdio.h>
#include <stdlib.h>

#define MAX 100000

typedef struct {
    int arr[MAX];
    int top;
} Stack;

void push(Stack *s, int val) {
    s->arr[++(s->top)] = val;
}

int pop(Stack *s) {
    return s->arr[(s->top)--];
}

int peek(Stack *s) {
    return s->arr[s->top];
}

int isEmpty(Stack *s) {
    return s->top == -1;
}

int main() {
    int q;
    scanf("%d", &q);

    Stack stack_in;
    Stack stack_out;
    stack_in.top = -1;
    stack_out.top = -1;

    while (q--) {
        int query;
        scanf("%d", &query);

        if (query == 1) {
            int x;
            scanf("%d", &x);
            push(&stack_in, x);
        } else if (query == 2) {
            if (isEmpty(&stack_out)) {
                while (!isEmpty(&stack_in)) {
                    push(&stack_out, pop(&stack_in));
                }
            }
            pop(&stack_out);
        } else if (query == 3) {
            if (isEmpty(&stack_out)) {
                while (!isEmpty(&stack_in)) {
                    push(&stack_out, pop(&stack_in));
                }
            }
            printf("%d\n", peek(&stack_out));
        }
    }

    return 0;
}
